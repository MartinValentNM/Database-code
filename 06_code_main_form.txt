' ===== SHARED HELPERS =====
Public Function SanitizeName(s As String) As String
    Dim t As String: t = s
    t = Replace(t, " ", "_"): t = Replace(t, "[", ""): t = Replace(t, "]", "")
    t = Replace(t, "/", "_"): t = Replace(t, "\", "_"): t = Replace(t, ":", "")
    t = Replace(t, "(", ""): t = Replace(t, ")", ""): t = Replace(t, ".", "")
    t = Replace(t, "-", "_"): t = Replace(t, "'", ""): t = Replace(t, Chr(34), "")
    t = Replace(t, Chr(10), ""): t = Replace(t, Chr(13), "")
    SanitizeName = t
End Function

Public Sub UpdateFieldInDB(tid As Long, fldName As String, newVal As String)
    On Error Resume Next
    Dim tbl As String
    Select Case UCase(fldName)
        Case "GENUS", "SPECIES", "FAMILY", "ORDER", "CLASS", "PHYLUM", "AUTHOR", "RANK", "GEOGRAPHY", "Formation", "CLEN", "STUPEN", "SERIE", "ZONA", "HORIZONT": tbl = "Taxa"
        Case "TAXONOMIC PLACEMENT", "TAXON", "TYPE TAXON", "INCLUDED TAXONS": tbl = "Taxa_Taxonomie"
        Case "ETYMOLOGY", "DESCRIPTION", "SIZE", "DIAGNOSIS", "FIGURES", "OPERCULUM", "APERTURE", "CROSS-SECTION", "SCULPTURE", "CLAVICLES", "CARDINAL_PROCESSES": tbl = "Taxa_Popis"
        Case "TYPE MATERIAL", "MATERIAL EXAMINED", "SYNONYMY": tbl = "Taxa_Material"
        Case "OCCURRENCE", "STRATIGRAPHY", "LOCALITY": tbl = "Taxa_Vyskyt"
        Case "REMARKS", "Reference": tbl = "Taxa_Poznamky"
        Case Else: tbl = "Taxa_Poznamky"
    End Select
    Dim whereF As String: whereF = IIf(tbl = "Taxa", "ID", "TaxonID")
    CurrentDb.Execute "UPDATE [" & tbl & "] SET [" & fldName & "] = '" & Replace(newVal, "'", "''") & "' WHERE [" & whereF & "] = " & tid, 128
End Sub

Public Function GetFieldList() As String
    On Error Resume Next
    Dim db As DAO.Database: Set db = CurrentDb
    Dim qd As DAO.QueryDef: Set qd = db.QueryDefs("vw_Complete_Taxa")
    If qd Is Nothing Then GetFieldList = "": Exit Function
    Dim result As String
    Dim f As DAO.Field
    For Each f In qd.fields
        If f.Name <> "ID" And f.Name <> "Compare" Then result = result & f.Name & "|"
    Next f
    GetFieldList = result
End Function

Private Function GetCompareFunction() As String
    Dim code As String
    code = "Private Sub btnCompare_Click()" & vbCrLf
    code = code & " On Error Resume Next" & vbCrLf
    code = code & " Dim ids() As Long, cnt As Integer: cnt = 0" & vbCrLf
    code = code & " ReDim ids(0 To 3)" & vbCrLf
    code = code & " Dim rs As Object: Set rs = Me.subResults.Form.Recordset.Clone" & vbCrLf
    code = code & " rs.MoveFirst" & vbCrLf
    code = code & " Do While Not rs.EOF And cnt < 4" & vbCrLf
    code = code & "   If rs.Fields(""Compare"") = True Then ids(cnt) = rs.Fields(""ID""): cnt = cnt + 1" & vbCrLf
    code = code & "   rs.MoveNext" & vbCrLf
    code = code & " Loop" & vbCrLf
    code = code & " If cnt < 2 Then MsgBox ""Select 2-4 records (Compare checkbox)."", vbInformation: Exit Sub" & vbCrLf
    code = code & " DoCmd.OpenForm ""frmCompare""" & vbCrLf
    code = code & " Dim cf As Form: Set cf = Forms(""frmCompare"")" & vbCrLf
    code = code & " Dim db As Object: Set db = CurrentDb" & vbCrLf
    code = code & " Dim container As Object" & vbCrLf
    code = code & " On Error Resume Next" & vbCrLf
    code = code & " Set container = db.QueryDefs(""vw_Complete_Taxa"")" & vbCrLf
    code = code & " If container Is Nothing Then Set container = db.TableDefs(""vw_Complete_Taxa"")" & vbCrLf
    code = code & " On Error GoTo 0" & vbCrLf
    code = code & " Dim i As Integer" & vbCrLf
    code = code & " For i = 1 To cnt" & vbCrLf
    code = code & "   Dim rsTax As Object: Set rsTax = db.OpenRecordset(""SELECT * FROM vw_Complete_Taxa WHERE ID="" & ids(i-1))" & vbCrLf
    code = code & "   If Not rsTax.EOF Then" & vbCrLf
    code = code & "     cf.Controls(""lblRec"" & i).Caption = Nz(rsTax.Fields(""TAXON""), ""ID="" & ids(i-1))" & vbCrLf
    code = code & "     cf.Controls(""lblRec"" & i).Visible = True" & vbCrLf
    code = code & "     Dim fld As Object" & vbCrLf
    code = code & "     For Each fld In container.Fields" & vbCrLf
    code = code & "       If UCase(fld.Name) <> ""COMPARE"" Then" & vbCrLf
    code = code & "         Dim sn As String: sn = SanitizeName(fld.Name)" & vbCrLf
    code = code & "         On Error Resume Next" & vbCrLf
    code = code & "         cf.Controls(""cmp_"" & sn & ""_"" & i).Value = Nz(rsTax.Fields(fld.Name), """")" & vbCrLf
    code = code & "         cf.Controls(""cmp_"" & sn & ""_"" & i).Visible = True" & vbCrLf
    code = code & "         On Error GoTo 0" & vbCrLf
    code = code & "       End If" & vbCrLf
    code = code & "     Next fld" & vbCrLf
    code = code & "   End If" & vbCrLf
    code = code & "   rsTax.Close" & vbCrLf
    code = code & " Next i" & vbCrLf
    code = code & "End Sub"
    GetCompareFunction = code
End Function

Public Sub RestoreTaxonFromMerge(tid As Long)
    ' Restores a taxon that was merged
    On Error Resume Next
    Dim db As DAO.Database: Set db = CurrentDb
    
    ' Check if taxon already exists (was deleted during merge)
    Dim rs As DAO.Recordset
    Set rs = db.OpenRecordset("SELECT COUNT(*) AS cnt FROM Taxa WHERE ID=" & tid)
    If rs!cnt > 0 Then
        rs.Close
        MsgBox "Taxon ID=" & tid & " already exists - skipping.", vbInformation
        Exit Sub
    End If
    rs.Close
    
    ' Restore from Taxa_History (ActionType=MERGE)
    Dim tables(5) As String
    Dim whereFields(5) As String
    tables(0) = "Taxa":           whereFields(0) = "ID"
    tables(1) = "Taxa_Taxonomie": whereFields(1) = "TaxonID"
    tables(2) = "Taxa_Popis":     whereFields(2) = "TaxonID"
    tables(3) = "Taxa_Material":  whereFields(3) = "TaxonID"
    tables(4) = "Taxa_Vyskyt":    whereFields(4) = "TaxonID"
    tables(5) = "Taxa_Poznamky":  whereFields(5) = "TaxonID"
    
    Dim i As Integer
    For i = 0 To 5
        Dim rsH As DAO.Recordset
        Set rsH = db.OpenRecordset("SELECT FieldName, OldValue FROM Taxa_History WHERE TaxonID=" & tid & " AND ActionType='MERGE' AND TableName='" & tables(i) & "' ORDER BY HistoryID")
        If Not rsH.EOF Then
            ' Create record
            If i = 0 Then
                db.Execute "INSERT INTO Taxa (ID) VALUES (" & tid & ")", 128
            Else
                Dim rsC As DAO.Recordset
                Set rsC = db.OpenRecordset("SELECT COUNT(*) AS cnt FROM [" & tables(i) & "] WHERE TaxonID=" & tid)
                If rsC!cnt = 0 Then db.Execute "INSERT INTO [" & tables(i) & "] (TaxonID) VALUES (" & tid & ")", 128
                rsC.Close
            End If
            
            ' Set field values
            Do While Not rsH.EOF
                Dim fName As String: fName = rsH!FieldName
                Dim fVal As String: fVal = Nz(rsH!OldValue, "")
                If fName <> "ID" And fName <> "TaxonID" Then
                    db.Execute "UPDATE [" & tables(i) & "] SET [" & fName & "] = '" & Replace(fVal, "'", "''") & "' WHERE [" & whereFields(i) & "] = " & tid, 128
                End If
                rsH.MoveNext
            Loop
        End If
        rsH.Close
    Next i
    
    ' Delete MERGE history for this taxon
    db.Execute "DELETE FROM Taxa_History WHERE TaxonID=" & tid & " AND ActionType='MERGE'", 128
End Sub
